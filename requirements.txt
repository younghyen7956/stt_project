# --- FastAPI Web Framework & Server ---
fastapi               # API 서버를 구축하기 위해 사용 (APIRouter, Depends, HTTPException 등)
uvicorn[standard]     # FastAPI 애플리케이션을 실행하는 ASGI 서버 (개발용 및 Gunicorn 워커로 사용)
gunicorn              # 프로덕션 환경에서 uvicorn 워커를 관리하는 WSGI 서버 (docker-compose.yml에서 사용)

# --- Environment & Utilities ---
python-dotenv         # .env 파일에서 환경 변수(API 키 등)를 로드하기 위해 사용
pycountry             # ISO 언어 코드를 전체 언어 이름으로 변환하기 위해 사용 (e.g., 'ko' -> 'Korean')
python-multipart

# --- AI / Machine Learning Models ---
openai                # OpenAI의 GPT 모델(gpt-4o-mini)을 사용하여 텍스트 번역 및 답변 생성을 위해 사용
openai-whisper        # 음성 파일의 언어를 감지하는 데 사용 (whisper.load_model("base"))
transformers          # Hugging Face의 STT 모델(whisper-large-v3-turbo)을 로드하고 사용하기 위한 파이프라인
torch                 # transformers와 whisper 모델을 실행하기 위한 핵심 딥러닝 프레임워크
kokoro            # 텍스트를 음성(TTS)으로 변환하기 위해 사용 (KPipeline)

# --- Audio Processing ---
pydub                 # 다양한 오디오 포맷(mp3, m4a 등)을 처리하고 표준 형식으로 변환하기 위해 사용
soundfile             # 오디오 데이터를 NumPy 배열로 읽고 쓰기 위해 사용 (주로 WAV 형식 처리)
numpy                 # 오디오 데이터를 수치 배열(array)로 다루기 위해 사용

# --- Data Handling ---
pandas                # 코드에 import 되어 있으므로 포함 (향후 데이터 분석/처리에 사용될 수 있음)
misaki[ja]
misaki[zh]
#misaki[ko]
